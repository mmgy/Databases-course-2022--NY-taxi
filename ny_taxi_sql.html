Next cell showcases the logic with which I solved the problem. I took the 500 km radius around the center of NY and examined the taxi traffic between concentric circular areas ranging up to 500 km. I converted the lat-lon coordinates to km by using a local flat earth approximation (this disregards the local curvature of Earth and essentially treats the lat-lon coordinates in a local Descartian coordinate system factored with the appropriate coefficients).

def get_OD_element(dist_from_idx, dist_to_idx, time_of_day = [6, 11]):
    sql_el = f"SELECT COUNT(pickup_datetime) FROM taxi.taxi_data WHERE (pickup_dist BETWEEN {dist_points[dist_from_idx]} AND {dist_points[dist_from_idx+1]}) AND (dropoff_dist BETWEEN {dist_points[dist_to_idx]} AND {dist_points[dist_to_idx+1]}) AND (HOUR(STR_TO_DATE(pickup_datetime, '%Y-%m-%d %T')) BETWEEN {time_of_day[0]} AND {time_of_day[1]});"
    cursor.execute(sql_el)
    result = cursor.fetchall()
    return result[0][0]

def get_OD_matrix(dist_points, times_of_day):
    ODs = []
    for time_of_day in times_of_day:
        OD = []
        for i in range(len(dist_points) - 1):
            OD_x = []
            for j in range(len(dist_points) - 1):
                element = get_OD_element(i, j, time_of_day=time_of_day)
                OD_x.append(element)
            OD.append(OD_x)
        ODs.append(OD)
    return np.array(ODs)

for k in range(OD_matrices.shape[0]):
    fig, ax = plt.subplots(figsize = (12, 12))
    mts = ax.matshow(OD_matrices[k][::-1,::-1].T, norm = colors.LogNorm(vmin = 1, vmax = OD_matrices[k].max()), cmap = plt.cm.jet)
    ax.set_xticks(np.arange(0, len(dist_points)-1))
    ax.set_xticklabels(dist_points[1:])
    ax.set_yticks(np.arange(0, len(dist_points)-1))
    ax.set_yticklabels(dist_points[1:])
    for i in range(len(dist_points)-1):
        for j in range(len(dist_points)-1):
            c = OD_matrices[k][::-1,::-1].T[j,i]
            ax.text(i, j, str(int(c)), va='center', ha='center', fontsize = 7)
    fig.colorbar(mts, ax=ax, extend = 'max')
    ax.set_title(f"NY taxi OD matrix 2013 January between {int(times_of_day[k][0])} and {int(times_of_day[k][1])+1} hours")
    ax.set_xlabel("Dropoff distance from NY center [km]")
    ax.set_ylabel("Pickup distance from NY center [km]")
    plt.savefig(f"od_matrix_{k}.png")
